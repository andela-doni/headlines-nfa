/**
 * Gets the cookie info and dispatches the LOGIN action
 * @argument {sortBy,source}.
 * @returns {nothing}.
 */
export function getArticles(source, sortBy) {
  request.get(url)
  .set('api', API)
  .query({ source })
  .query({ sortBy })
  .query({ apiKey: API })
  .end(articlesCallback);
}
